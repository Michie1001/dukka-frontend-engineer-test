App.js:

import {React, useState, useEffect} from "react";
import './App.css';
import Header from './components/Header';
import Create from './components/CreateEmployee';
import List from './components/ListEmployee';

function App() {

  const initialState = JSON.parse(localStorage.getItem("employees")) || []; //saving to local storage so we have the values even after reloading
  const [input, setInput] = useState("");
  const [employees, setEmployees] = useState(initialState);

  useEffect(() => {
    localStorage.setItem("employees", JSON.stringify(employees));
  }, [employees]);
  return (
    <div className="main">
      <Header />
      <div className="main-content">
        <Create
          input={input}
          setInput={setInput}
          employees={employees}
          setEmployees={setEmployees}
        />
        <List
          employees={employees}
          setEmployees={setEmployees}
        />
      </div>
    </div>
  );
}

export default App;

const List = ({employees, setEmployees}) => {

import React from 'react';
import {v4 as uuidv4} from "uuid"; //for unique ids

const Create = ({input, setInput, employees, setEmployees}) => {
	const onInputChange = (event) => {
		setInput(event.target.value);
	};
	const onFormSubmit = (event) => {
		event.preventDefault();
		setEmployees([...employees, {id: uuidv4(), title: input, completed: false}]);
		setInput("");
	};
	return (
			<div className="form">
				<h2>Create Employee Profile</h2>
				<form onSubmit={onFormSubmit}>
					<fieldset>
						<label>Full Name *</label>
						<input type="text" required value={input} onChange={onInputChange} />
					</fieldset>
					<fieldset>
					  <label>Phone Number *</label>
					  <input type="number" required />
					</fieldset>

					<button className="full-width-btn" type="submit">Save</button>
				</form>
				<button className="full-width-btn next">
					Next
					<i className="fas fa-long-arrow-alt-right"></i>
				</button>
			</div>
		);
};

export default Create;
